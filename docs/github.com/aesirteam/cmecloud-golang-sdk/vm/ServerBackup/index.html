type ServerBackupInterface interface {

    ServerBackupOrder() (string, error)

    // 创建云主机备份

    // 查看云主机备份列表

    // 使用云主机备份恢复云主机

    // 删除云主机备份
    DeleteServerBackup(backupId string) error
}
